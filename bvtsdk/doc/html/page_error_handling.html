Error Handling for Robust Code

The previous examples elided any error handling for the sake of clarity. You should not copy and paste those examples into production code!

The following code shows the significant amount of error checking you must perform if you use the C API. Manual memory management must be performed carefully.

Note
We strongly recommend you use the C++ API to avoid most of these issues.

#include <bvt_sdk.h>
#include <stdio.h>
#define IF_FAILED( expr, ret ) if ((ret = (expr)) != BVT_SUCCESS)
#define LOG_ERROR( ret ) printf("%s:%d: error '%s': '%s'\n", \
__FILE__, (int)__LINE__, \
BVTError_GetName(ret), \
BVTError_GetString(ret));
int main()
{
RetVal ret = BVT_SUCCESS;
BVTSonar sonar = BVTSonar_Create();
BVTHead head = NULL;
BVTPing ping = NULL;
IF_FAILED( BVTSonar_Open(sonar, "NET", "192.168.1.45"), ret )
{
LOG_ERROR(ret);
goto cleanup_have_sonar;
}
IF_FAILED( BVTSonar_GetHead(sonar, 0, &head), ret )
{
LOG_ERROR(ret);
goto cleanup_have_sonar; /* NB: Head is still NULL */
}
IF_FAILED( BVTHead_GetPing(head, -1, &ping), ret )
{
LOG_ERROR(ret);
goto cleanup_have_head; /* NB: Ping is still NULL */
}
/* Do something useful with the ping */
BVTPing_Destroy(ping);
cleanup_have_head:
BVTHead_Destroy(head);
cleanup_have_sonar:
BVTSonar_Destroy(sonar);
return ret;
}
